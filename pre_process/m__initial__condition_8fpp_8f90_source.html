160 !> This subroutine peruses the patches and depending on the
161 !! type of geometry associated with a particular patch, it
162 !! calls the related subroutine to setup the said geometry
163 !! on the grid using the primitive variables included with
164 !! the patch parameters. The subroutine is complete once the
165 !! primitive variables are converted to conservative ones.

167
168 integer :: i

170 ! Converting the conservative variables to the primitive ones given
171 ! preexisting initial condition data files were read in on start-up
172 if (old_ic) then
173 call s_convert_conservative_to_primitive_variables(q_cons_vf, &
174 q_t_sf, &
175 q_prim_vf, &
176 idwbuff)
177 end if
178
179 call s_apply_icpp_patches(patch_id_fp, q_prim_vf)
180
181 if (num_bc_patches > 0) call s_apply_boundary_patches(q_prim_vf, bc_type)
182
183 if (perturb_flow) call s_perturb_surrounding_flow(q_prim_vf)

185 if (mixlayer_perturb) call s_perturb_mixlayer(q_prim_vf)
186 if (simplex_perturb) call s_perturb_simplex(q_prim_vf)
187 if (elliptic_smoothing) call s_elliptic_smoothing(q_prim_vf, bc_type)
188
189 ! Converting the primitive variables to the conservative ones
190 call s_convert_primitive_to_conservative_variables(q_prim_vf, q_cons_vf)
191
192 if (chemistry) call s_compute_t_from_primitives(q_t_sf, q_prim_vf, idwint)
193
194 if (qbmm .and. .not. polytropic) then
195 !Initialize pb and mv
196 call s_initialize_mv(q_cons_vf, mv%sf)

198 end if
199
200 end subroutine s_generate_initial_condition
