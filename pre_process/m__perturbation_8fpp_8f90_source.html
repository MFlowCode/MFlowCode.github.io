193 end if

231 !> Compute velocity perturbations for a temporal mixing layer with a hyperbolic tangent mean streamwise velocity profile, using
232 !! an inverted version of the spectrum-based synthetic turbulence generation method proposed by Guo et al. (2023, JFM).
233 subroutine s_perturb_mixlayer(q_prim_vf)
234
235 type(scalar_field), dimension(sys_size), intent(inout) :: q_prim_vf
236 real(wp), dimension(mixlayer_perturb_nk) :: k, Ek
237 real(wp), dimension(3, 3) :: Rij, Lmat
238 real(wp), dimension(3) :: velfluc, sig_tmp, sig, khat, xi
239 real(wp) :: dk, alpha, Eksum, q, uu0, phi
240 integer :: i, j, l, r, ierr
241
242 dk = 1._wp/mixlayer_perturb_nk
243
244 ! Compute prescribed energy spectra
245 eksum = 0._wp
246 do i = 1, mixlayer_perturb_nk
247 k(i) = dk*i
248 ek(i) = (k(i)/mixlayer_perturb_k0)**4._wp*exp(-2._wp*(k(i)/mixlayer_perturb_k0)**2._wp)
249 eksum = eksum + ek(i)
250 end do
251
252 ! Main loop
253 do r = 0, n
254 ! Compute prescribed Reynolds stress tensor with about half magnitude of its self-similar value
255 rij(:,:) = 0._wp
256 uu0 = patch_icpp(1)%vel(1)**2._wp*(1._wp - tanh(y_cc(r)*mixlayer_vel_coef)**2._wp)
257 rij(1, 1) = 0.05_wp*uu0
258 rij(2, 2) = 0.03_wp*uu0
259 rij(3, 3) = 0.03_wp*uu0
260 rij(1, 2) = -0.02_wp*uu0
261 rij(2, 1) = rij(1, 2)
262
263 ! Cholesky decomposition for inhomogeneity and anisotropy
264 lmat = 0._wp
265 lmat(1, 1) = sqrt(rij(1, 1))
266 if (abs(lmat(1, 1)) < sgm_eps) lmat(1, 1) = sgm_eps
267 lmat(2, 1) = rij(2, 1)/lmat(1, 1)
268 lmat(2, 2) = sqrt(rij(2, 2) - lmat(2, 1)**2._wp)
269 if (abs(lmat(2, 2)) < sgm_eps) lmat(2, 2) = sgm_eps
270 lmat(3, 1) = rij(3, 1)/lmat(1, 1)
271 lmat(3, 2) = (rij(3, 2) - lmat(3, 1)*lmat(2, 1))/lmat(2, 2)
272 lmat(3, 3) = sqrt(rij(3, 3) - lmat(3, 1)**2._wp - lmat(3, 2)**2._wp)
273
274 ! Compute perturbation for each Fourier component
275 do i = 1, mixlayer_perturb_nk
276 ! Generate random numbers for unit wavevector khat, random unit vector xi, and random mode phase phi
277 if (proc_rank == 0) then
278 call s_generate_random_perturbation(khat, xi, phi, i, y_cc(r))
279 end if
280
281#ifdef MFC_MPI
282 call mpi_bcast(khat, 3, mpi_p, 0, mpi_comm_world, ierr)
283 call mpi_bcast(xi, 3, mpi_p, 0, mpi_comm_world, ierr)
284 call mpi_bcast(phi, 1, mpi_p, 0, mpi_comm_world, ierr)
285#endif
286
287 ! Compute mode direction by two-time cross product
288 sig_tmp = f_cross(xi, khat)
289 sig_tmp = sig_tmp/sqrt(sum(sig_tmp**2._wp))
290 sig = f_cross(khat, sig_tmp)
291
292 ! Compute perturbation for each grid
293 do l = 0, p
294 do j = 0, m
295 q = sqrt(ek(i)/eksum)
296 alpha = k(i)*(khat(1)*x_cc(j) + khat(2)*y_cc(r) + khat(3)*z_cc(l)) + 2._wp*pi*phi
297 velfluc = 2._wp*q*sig*cos(alpha)
298 velfluc = matmul(lmat, velfluc)
299 q_prim_vf(momxb)%sf(j, r, l) = q_prim_vf(momxb)%sf(j, r, l) + velfluc(1)
300 q_prim_vf(momxb + 1)%sf(j, r, l) = q_prim_vf(momxb + 1)%sf(j, r, l) + velfluc(2)
301 q_prim_vf(momxb + 2)%sf(j, r, l) = q_prim_vf(momxb + 2)%sf(j, r, l) + velfluc(3)
302 end do
303 end do
304 end do
305 end do
306
307 end subroutine s_perturb_mixlayer

310 subroutine s_generate_random_perturbation(khat, xi, phi, ik, yloc)
311
